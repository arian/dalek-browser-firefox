/**
 * Screenshot related WebDriver endpoints
 * see [JsonWireProtocol](https://code.google.com/p/selenium/wiki/JsonWireProtocol)
 *
 * @module FirefoxDriver
 * @class Screenshot
 * @namespace FirefoxDriver.Commands.WebDriver
 */

module.exports = function (Driver) {

  /**
   * Take a screenshot of the current page.
   *
   * @method screenshot
   * @see https://code.google.com/p/selenium/wiki/JsonWireProtocol#/session/:sessionId/screenshot
   * @param {GET} sessionId ID of the session to route the command to
   */

  Driver.addCommand({
    name: 'screenshot',
    url: '/session/:sessionId/screenshot',
    method: 'get'
  });

};
